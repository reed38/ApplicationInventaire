305 private void FindNextNoPresent()
306 {
307 ImageReleve = null;
308 ImageReleveName.Source = null;
309
310 if (IndiceSection == templateData.mySections.Count - 1 && IndicePiece == templateData.mySections[IndiceSection].PiecesList.Count)//if the user wen through everything

312 SaveAndQuit();

314
315 }
316
317 else if (IndicePiece == templateData.mySections[IndiceSection].PiecesList.Count) //if the user went throuht all the Pieces in the current Section
318 {
319 IndiceSection++;
320 IndicePiece = 0;
321 foreach (ImageInfos im in this.templateData.ImageSectionList)
322 {
323 if (im.Name == templateData.mySections[IndiceSection].SectionName)

327

336 {

384 SaveFileDialog saveFileDialog = new SaveFileDialog(); //opening windows popup to ask the user where he wants to save the resulting excel file.

386 // Set the default file name and extension
387 saveFileDialog.DefaultExt = ".xls";
388
389 // Set the file filters
390 saveFileDialog.Filter = "excel file (*.xls )|*.xls|excel file (*.xlsx)|*.xlsx"; // Set allowed file extensions
391
392 saveFileDialog.FileName = templateData.myTemplateInfos.TemplateName; // Set the default filename here
393 // Show the save file dialog
394 bool? result = saveFileDialog.ShowDialog();
395
396 // Process the selected file
397 if (result == true)

399 string filePath = saveFileDialog.FileName;

402 File.Copy(templateData.myTemplateInfos.TmpExcelPath, filePath, true);
403 }
