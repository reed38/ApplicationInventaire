156 private bool CheckAllIsHere()
157 {
158 bool res = true;
159
160 if (string.IsNullOrEmpty(TextBoxDescription.Text))
161 {
162 res = false;
163 POPUP.ShowPopup("Veillez saisir une descirption");
164 }
165 if (string.IsNullOrEmpty(TextBoxName.Text))
166 {
167 res = false;
168 POPUP.ShowPopup("Veuillez saisir le nom du template");
169 }
170 if (string.IsNullOrEmpty(excelPath))
171 {
172 res = false;
173 POPUP.ShowPopup("Veuillez sélectionner un fichier excel");
174 }
175 if (imageSectionsInfos.Count == 0)
176 {
177 res = false;
178 POPUP.ShowPopup("Veuillez sélectionner au moins une image de section");
179 }
180 if (PlansNameList.Count == 0)
181 {
182 res = false;
183 POPUP.ShowPopup("veuillez sélectionner au moins 1 plan");
184 }
185
186
187
188
189 string[] projectList = GlobalProjectData.GetProjectNames();
190 foreach (string project in projectList)
191 {
192 if (project.Equals(TextBoxName.Text))
193 {
194 res = false;
195 POPUP.ShowPopup("This name is already used ");
196 break;
197 }
198 }
