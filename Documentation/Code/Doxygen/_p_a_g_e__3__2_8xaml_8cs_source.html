275 private void FindNextNoPresent()
276 {
277 ImageReleve = null;
278 ImageReleveName.Source = null;
279
280 if (IndiceSection == projectData.mySections.Count - 1 && IndicePiece == projectData.mySections[IndiceSection].PiecesList.Count)//if the user wen through everything
281 {
282 SaveAndQuit();
283 return;
284
285 }
286
287 else if (IndicePiece == projectData.mySections[IndiceSection].PiecesList.Count) //if the user went throuht all the Pieces in the current Section
288 {
289 IndiceSection++;
290 IndicePiece = 0;
291 foreach (ImageInfos im in this.projectData.ImageSectionList)
292 {
293 if (im.Name == projectData.mySections[IndiceSection].SectionName)
294 {
295 ImageSection = im.Path; //updating Section image on te page
296 break;
297
298 }
299
300 }
301
302 }
303
304
305 for (int section = IndiceSection; section < projectData.mySections.Count; section++)
306 {
307
308 for (int piece = IndicePiece; piece < projectData.mySections[section].PiecesList.Count; piece++)
309 {
310
311 if (projectData.mySections[section].PiecesList[piece].IsPresent == 0)
312 {
313 this.IndicePiece = piece;
314 this.IndiceSection = section;
315 UpdateCurrent();
316 if (this.CurrentPiece.IsReleveRequired == 1)
317 {
318 ShowTextBoxSerialNumberConstructor();
319 foreach (ImageInfos i in projectData.ImageReleveList)
320 {
321 if (i.Name.IndexOf(this.CurrentPiece.PieceName)>=0)
322 {
323 ImageReleve = i.Path;
324 break;
325 }
326
327 }
328 }

330 return;
