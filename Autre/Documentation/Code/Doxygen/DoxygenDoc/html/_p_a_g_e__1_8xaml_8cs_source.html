145 private void CopyDirectory(string sourceDirectory, string destinationDirectory)
146 {
147 // Create the destination directory if it doesn't exist
148 Directory.CreateDirectory(destinationDirectory);
149
150 // Get the files in the source directory
151 string[] files = Directory.GetFiles(sourceDirectory);
152
153 // Copy each file to the destination directory
154 foreach (string file in files)
155 {
156 string fileName = System.IO.Path.GetFileName(file);
157 string destinationPath = System.IO.Path.Combine(destinationDirectory, fileName);
158 File.Copy(file, destinationPath, true);
159 }
160
161 // Recursively copy subdirectories
162 string[] directories = Directory.GetDirectories(sourceDirectory);
163 foreach (string directory in directories)
164 {
165 string directoryName = System.IO.Path.GetFileName(directory);
166 string destinationPath = System.IO.Path.Combine(destinationDirectory, directoryName);
167 CopyDirectory(directory, destinationPath);
168 }
