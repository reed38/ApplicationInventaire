111 {
112 string fileExtension = ".pdf";
113 // Get the file class key associated with the file extension
114 string fileClassKey = Registry.ClassesRoot.OpenSubKey(fileExtension)?.GetValue("")?.ToString();
115
116 if (fileClassKey != null)
117 {
118 // Get the default application command associated with the file class

121 if (defaultApplicationCommand != null)

123 // Extract the executable path from the command string
